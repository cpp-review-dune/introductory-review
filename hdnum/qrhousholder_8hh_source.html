51  auto m = A.rowsize();
52  auto n = A.colsize();
53  for (size_t j = 0; j < n; j++) {
54  REAL s = 0;
55  for (size_t i = j; i < m; i++) {
56  s = s + pow(A(i, j), 2);
57  }
58  s = sqrt(s);
59  v[j] = (-1.0) * sgn(A(j, j)) * s;
60  REAL fak = sqrt(s * (s + std::abs(A(j, j))));
61  A(j, j) = A(j, j) - v[j];
62  for (size_t k = j; k < m; k++) {
63  A(k, j) = A(k, j) / fak;
64  }

66  s = 0;
